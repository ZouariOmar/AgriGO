1359 public static function validateAddress($address, $patternselect = null)
1360 {
1361 if (null === $patternselect) {
1362 $patternselect = static::$validator;
1363 }
1364 //Don't allow strings as callables, see SECURITY.md and CVE-2021-3603
1365 if (is_callable($patternselect) && !is_string($patternselect)) {
1366 return call_user_func($patternselect, $address);
1367 }
1368 //Reject line breaks in addresses; it's valid RFC5322, but not RFC5321
1369 if (strpos($address, "\n") !== false || strpos($address, "\r") !== false) {
1370 return false;
1371 }
1372 switch ($patternselect) {
1373 case 'pcre': //Kept for BC
1374 case 'pcre8':
1375 /*
1376 * A more complex and more permissive version of the RFC5322 regex on which FILTER_VALIDATE_EMAIL
1377 * is based.
1378 * In addition to the addresses allowed by filter_var, also permits:
1379 * * dotless domains: `a@b`
1380 * * comments: `1234 @ local(blah) .machine .example`
1381 * * quoted elements: `'"test blah"@example.org'`
1382 * * numeric TLDs: `a@b.123`
1383 * * unbracketed IPv4 literals: `a@192.168.0.1`
1384 * * IPv6 literals: 'first.last@[IPv6:a1::]'
1385 * Not all of these will necessarily work for sending!
1386 *
1387 * @see http://squiloople.com/2009/12/20/email-address-validation/
1388 * @copyright 2009-2010 Michael Rushton
1389 * Feel free to use and redistribute this code. But please keep this copyright notice.
1390 */
1391 return (bool) preg_match(
1392 '/^(?!(?>(?1)"?(?>\\\[ -~]|[^"])"?(?1)){255,})(?!(?>(?1)"?(?>\\\[ -~]|[^"])"?(?1)){65,}@)' .
1393 '((?>(?>(?>((?>(?>(?>\x0D\x0A)?[\t ])+|(?>[\t ]*\x0D\x0A)?[\t ]+)?)(\‍((?>(?2)' .
1394 '(?>[\x01-\x08\x0B\x0C\x0E-\'*-\[\]-\x7F]|\\\[\x00-\x7F]|(?3)))*(?2)\‍)))+(?2))|(?2))?)' .
1395 '([!#-\'*+\/-9=?^-~-]+|"(?>(?2)(?>[\x01-\x08\x0B\x0C\x0E-!#-\[\]-\x7F]|\\\[\x00-\x7F]))*' .
1396 '(?2)")(?>(?1)\.(?1)(?4))*(?1)@(?!(?1)[a-z0-9-]{64,})(?1)(?>([a-z0-9](?>[a-z0-9-]*[a-z0-9])?)' .
1397 '(?>(?1)\.(?!(?1)[a-z0-9-]{64,})(?1)(?5)){0,126}|\[(?:(?>IPv6:(?>([a-f0-9]{1,4})(?>:(?6)){7}' .
1398 '|(?!(?:.*[a-f0-9][:\]]){8,})((?6)(?>:(?6)){0,6})?::(?7)?))|(?>(?>IPv6:(?>(?6)(?>:(?6)){5}:' .
1399 '|(?!(?:.*[a-f0-9]:){6,})(?8)?::(?>((?6)(?>:(?6)){0,4}):)?))?(25[0-5]|2[0-4][0-9]|1[0-9]{2}' .
1400 '|[1-9]?[0-9])(?>\.(?9)){3}))\])(?1)$/isD',
1401 $address
1402 );
1403 case 'html5':
1404 /*
1405 * This is the pattern used in the HTML5 spec for validation of 'email' type form input elements.
1406 *
1407 * @see https://html.spec.whatwg.org/#e-mail-state-(type=email)
1408 */
1409 return (bool) preg_match(
1410 '/^[a-zA-Z0-9.!#$%&\'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}' .
1411 '[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/sD',
1412 $address
1413 );
1414 case 'php':
1415 default:
1416 return filter_var($address, FILTER_VALIDATE_EMAIL) !== false;
1417 }

1807 protected static function isShellSafe($string)
1808 {

1813 return false;
1814 }

1817 escapeshellcmd($string) !== $string
1818 || !in_array(escapeshellarg($string), ["'$string'", "\"$string\""])

1831 if (!ctype_alnum($c) && strpos('@_-.', $c) === false) {
1832 return false;
1833 }
1834 }

2520 public function utf8CharBoundary($encodedText, $maxLength)
2521 {
2522 $foundSplitPos = false;
2523 $lookBack = 3;
2524 while (!$foundSplitPos) {
2525 $lastChunk = substr($encodedText, $maxLength - $lookBack, $lookBack);
2526 $encodedCharPos = strpos($lastChunk, '=');
2527 if (false !== $encodedCharPos) {
2528 //Found start of encoded character byte within $lookBack block.
2529 //Check the encoded byte value (the 2 chars after the '=')
2530 $hex = substr($encodedText, $maxLength - $lookBack + $encodedCharPos + 1, 2);
2531 $dec = hexdec($hex);
2532 if ($dec < 128) {
2533 //Single byte character.
2534 //If the encoded char was found at pos 0, it will fit
2535 //otherwise reduce maxLength to start of the encoded char
2536 if ($encodedCharPos > 0) {
2537 $maxLength -= $lookBack - $encodedCharPos;
2538 }
2539 $foundSplitPos = true;
2540 } elseif ($dec >= 192) {
2541 //First byte of a multi byte character
2542 //Reduce maxLength to split at start of character
2543 $maxLength -= $lookBack - $encodedCharPos;
2544 $foundSplitPos = true;
2545 } elseif ($dec < 192) {
2546 //Middle byte of a multi byte character, look further back
2547 $lookBack += 3;
2548 }
2549 } else {
2550 //No encoded character found
2551 $foundSplitPos = true;
2552 }
2553 }

4749 public function DKIM_HeaderC($signHeader)
4750 {
4751 //Normalize breaks to CRLF (regardless of the mailer)
4752 $signHeader = static::normalizeBreaks($signHeader, self::CRLF);
4753 //Unfold header lines
4754 //Note PCRE \s is too broad a definition of whitespace; RFC5322 defines it as `[ \t]`
4755 //@see https://tools.ietf.org/html/rfc5322#section-2.2
4756 //That means this may break if you do something daft like put vertical tabs in your headers.
4757 $signHeader = preg_replace('/\r\n[ \t]+/', ' ', $signHeader);
4758 //Break headers out into an array
4759 $lines = explode(self::CRLF, $signHeader);
4760 foreach ($lines as $key => $line) {
4761 //If the header is missing a :, skip it as it's invalid
4762 //This is likely to happen because the explode() above will also split
4763 //on the trailing LE, leaving an empty line
4764 if (strpos($line, ':') === false) {
4765 continue;
4766 }
4767 list($heading, $value) = explode(':', $line, 2);
4768 //Lower-case header name
4769 $heading = strtolower($heading);
4770 //Collapse white space within the value, also convert WSP to space
4771 $value = preg_replace('/[ \t]+/', ' ', $value);
4772 //RFC6376 is slightly unclear here - it says to delete space at the *end* of each value
4773 //But then says to delete space before and after the colon.
4774 //Net result is the same as trimming both ends of the value.
4775 //By elimination, the same applies to the field name
4776 $lines[$key] = trim($heading, " \t") . ':' . trim($value, " \t");

4819 $DKIMcanonicalization = 'relaxed/simple'; //Canonicalization methods of header & body
4820 $DKIMquery = 'dns/txt'; //Query method
4821 $DKIMtime = time();
4822 //Always sign these headers without being asked
4823 //Recommended list from https://tools.ietf.org/html/rfc6376#section-5.4.1

4826 'to',
4827 'cc',
4828 'date',
4829 'subject',
4830 'reply-to',
4831 'message-id',
4832 'content-type',
4833 'mime-version',
4834 'x-mailer',

4837 $headers_line .= 'Subject: ' . $subject . static::$LE;

4841 $currentHeaderValue = '';
4842 $parsedHeaders = [];

4844 $headerLineCount = count($headerLines);

4846 $matches = [];

4849 //We were previously in another header; This is the start of a new header, so save the previous one
4850 $parsedHeaders[] = ['label' => $currentHeaderLabel, 'value' => $currentHeaderValue];
4851 }
4852 $currentHeaderLabel = $matches[1];
4853 $currentHeaderValue = $matches[2];
4854 } elseif (preg_match('/^[ \t]+(.*)$/', $headerLine, $matches)) {
4855 //This is a folded continuation of the current header, so unfold it
4856 $currentHeaderValue .= ' ' . $matches[1];
4857 }

4859 if ($headerLineIndex >= $headerLineCount) {
4860 //This was the last line, so finish off this header
4861 $parsedHeaders[] = ['label' => $currentHeaderLabel, 'value' => $currentHeaderValue];
4862 }
4863 }

4865 $headersToSignKeys = [];
4866 $headersToSign = [];

4868 //Is this header one that must be included in the DKIM signature?

4870 $headersToSignKeys[] = $header['label'];
4871 $headersToSign[] = $header['label'] . ': ' . $header['value'];
4872 if ($this->DKIM_copyHeaderFields) {
4873 $copiedHeaders[] = $header['label'] . ':' . //Note no space after this, as per RFC
4874 str_replace('|', '=7C', $this->DKIM_QP($header['value']));
4875 }
4876 continue;
4877 }
4878 //Is this an extra custom header we've been asked to sign?
4879 if (in_array($header['label'], $this->DKIM_extraHeaders, true)) {
4880 //Find its value in custom headers
4881 foreach ($this->CustomHeader as $customHeader) {
4882 if ($customHeader[0] === $header['label']) {
4883 $headersToSignKeys[] = $header['label'];
4884 $headersToSign[] = $header['label'] . ': ' . $header['value'];
4885 if ($this->DKIM_copyHeaderFields) {
4886 $copiedHeaders[] = $header['label'] . ':' . //Note no space after this, as per RFC
4887 str_replace('|', '=7C', $this->DKIM_QP($header['value']));
4888 }
4889 //Skip straight to the next header
4890 continue 2;
4891 }
4892 }
4893 }
4894 }
4895 $copiedHeaderFields = '';
4896 if ($this->DKIM_copyHeaderFields && count($copiedHeaders) > 0) {
4897 //Assemble a DKIM 'z' tag
4898 $copiedHeaderFields = ' z=';

4902 $copiedHeaderFields .= static::$LE . ' |';

4908 0,
4909 -strlen(static::$LE . self::FWS)

4913 }

4921 //Base64 of packed binary SHA-256 hash of body
4922 $DKIMb64 = base64_encode(pack('H*', hash('sha256', $body)));
4923 $ident = '';

4925 $ident = ' i=' . $this->DKIM_identity . ';' . static::$LE;
4926 }

4928 //which is appended after calculating the signature
4929 //https://tools.ietf.org/html/rfc6376#section-3.5

4935 ' t=' . $DKIMtime . ';' .
4936 ' c=' . $DKIMcanonicalization . ';' . static::$LE .

4938 $ident .
4939 $copiedHeaderFields .
4940 ' bh=' . $DKIMb64 . ';' . static::$LE .
4941 ' b=';

4944 $headerValues . static::$LE . $dkimSignatureHeader
4945 );
